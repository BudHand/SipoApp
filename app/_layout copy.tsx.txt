// app/_layout.tsx
import BottomNav from "@/components/BottomNav";
import { useColorScheme } from "@/hooks/use-color-scheme";
import {
	Inter_400Regular,
	Inter_500Medium,
	Inter_600SemiBold,
	Inter_700Bold,
	useFonts,
} from "@expo-google-fonts/inter";
import { DarkTheme, DefaultTheme, ThemeProvider } from "@react-navigation/native";
import * as Notifications from "expo-notifications";
import { Stack, usePathname } from "expo-router";
import { StatusBar } from "expo-status-bar";
import { useEffect } from "react";
import { View } from "react-native";

export const unstable_settings = { anchor: "login" };

const groupsWithBottomNav = [
  "/beranda",
  "/notifikasi",
  "/memo",
  "/profil",
  "/undangan",
  "/approval",
  "/risalah",
];

export default function RootLayout() {
  const [fontsLoaded] = useFonts({
    Inter_400Regular,
    Inter_500Medium,
    Inter_600SemiBold,
    Inter_700Bold,
  });
  const colorScheme = useColorScheme();
  const pathname = usePathname();
  // removed unused params to avoid linter/TS warnings

  useEffect(() => {
    // listener untuk notifikasi diterima saat app foreground
    const notificationListener = Notifications.addNotificationReceivedListener((notification) => {
      console.log("📩 Notifikasi diterima (foreground):", notification);
    });

    // listener untuk ketika user menekan notifikasi
    const responseListener = Notifications.addNotificationResponseReceivedListener((response) => {
      console.log("👆 User menekan notifikasi:", response);
      // contoh navigasi bisa dilakukan di sini (pakai router.push jika diperlukan)
    });

    // cleanup: gunakan .remove() pada object subscription
    return () => {
      try {
        notificationListener.remove();
      } catch (e) {
        // fallback: jika remove tidak tersedia, ignore
      }
      try {
        responseListener.remove();
      } catch (e) {
        // fallback: jika remove tidak tersedia, ignore
      }
    };
  }, []);

  if (!fontsLoaded) return null;

  const showBottomNav = groupsWithBottomNav.some((p) => pathname.startsWith(p));

  return (
    <ThemeProvider value={colorScheme === "dark" ? DarkTheme : DefaultTheme}>
      <View style={{ flex: 1 }}>
        <Stack initialRouteName="splash/splash1">
          {/* screens */}
          <Stack.Screen name="beranda/beranda" options={{ headerShown: false }} />
          <Stack.Screen name="notifikasi/notifikasi" options={{ headerShown: false }} />
          <Stack.Screen name="splash/splash1" options={{ headerShown: false }} />
          <Stack.Screen name="splash/splash2" options={{ headerShown: false }} />
          <Stack.Screen name="splash/splash3" options={{ headerShown: false }} />
          <Stack.Screen name="splash/splash4" options={{ headerShown: false }} />
          <Stack.Screen name="login" options={{ headerShown: false }} />
          <Stack.Screen name="modal" options={{ presentation: "modal", title: "Modal" }} />
          <Stack.Screen name="memo/memos" options={{ headerShown: false }} />
          <Stack.Screen name="memo/memo-detail" options={{ headerShown: false }} />
          <Stack.Screen name="undangan/undangan" options={{ headerShown: false }} />
          <Stack.Screen name="undangan/undangan-detail" options={{ headerShown: false }} />
          <Stack.Screen name="risalah/risalah" options={{ headerShown: false }} />
          <Stack.Screen name="risalah/risalah-detail" options={{ headerShown: false }} />
          <Stack.Screen name="approval/approval" options={{ headerShown: false }} />
          <Stack.Screen name="profil/profil" options={{ headerShown: false }} />
        </Stack>

        <StatusBar style="auto" />

        {showBottomNav && (
          <View
            pointerEvents="box-none"
            style={{ position: "absolute", left: 0, right: 0, bottom: 0 }}
          >
            <BottomNav />
          </View>
        )}
      </View>
    </ThemeProvider>
  );
}
