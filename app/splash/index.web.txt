import Dots from "@/components/Dots";
import { Colors } from "@/constants/colors";
import { Stack, useRouter } from "expo-router";
import React, { useState } from "react";
import { ScrollView, StyleSheet, Text, View } from "react-native";
import Splash2 from "./splash2";
import Splash3 from "./splash3";
import Splash4 from "./splash4";

export default function SplashPagerWeb() {
  const [page] = useState(0);
  const router = useRouter();

  React.useEffect(() => {
    const timer = setTimeout(() => router.replace("/login"), 1500);
    return () => clearTimeout(timer);
  }, [router]);

  return (
    <View style={{ flex: 1, backgroundColor: Colors.white }}>
      <Stack.Screen options={{ headerShown: false }} />
      <ScrollView
        style={styles.webContainer}
        showsVerticalScrollIndicator={false}
      >
        <Splash2 />
        <Splash3 />
        <Splash4 />
        <View style={styles.lastSlide}>
          <View style={styles.transitionCircle} />
          <Text style={{ color: Colors.white, marginTop: 10 }}>
            Web Preview Mode
          </Text>
        </View>
      </ScrollView>

      <View style={styles.dotsContainer}>
        <Dots
          total={4}
          activeIndex={page}
          color={Colors.navy}
          inactiveColor="rgba(30,65,120,0.2)"
          size={6}
          spacing={6}
        />
      </View>
    </View>
  );
}

const styles = StyleSheet.create({
  webContainer: { flex: 1 },
  dotsContainer: {
    position: "absolute",
    bottom: 90,
    width: "100%",
    alignItems: "center",
  },
  lastSlide: {
    flex: 1,
    backgroundColor: Colors.navy,
    justifyContent: "center",
    alignItems: "center",
    paddingVertical: 100,
  },
  transitionCircle: {
    width: 120,
    height: 120,
    borderRadius: 60,
    backgroundColor: Colors.white,
    opacity: 0.4,
  },
});
